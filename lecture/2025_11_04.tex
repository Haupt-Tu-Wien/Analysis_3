In der Letzten Vorlesung haben wir den Satz von Stone-Weierstraß behandelt.
Zur Wiederholung: Sei $\mathcal{A}$ eine Unteralgebra von $C(X,\mathbb{R})$ 
die punktetrennend ist und $1 \in \mathcal{A}$, dann gilt 
$\overline{\mathcal{A}}^{\infty} = C(X,\mathbb{R})$.

\ex{}{$X:=[0,1], \mathcal{A} = \mathbb{R}[z]$}
Satz folgt wen wir den Spezialfall $1 \in \mathcal{A}$ beweisen Können.

Wir haben auch das Lemma \textbf{9.0.7} angefangen zu beweisen, 
zur Wiederholung die Aussage: \\
Sei $\mathcal{A} \subseteq C(X,\mathbb{R})$ Punktetrennende Unteralgebra mit $1 \in \mathcal{A}$, $A \subseteq X$ Abgeschlossen, $z \in X\setminus A$. Dann gilt:
Sei $\mathcal{A}\subseteq C(X,\mathbb{R})$ Punktetrennende Unteralgebra
mit $1 \in \mathcal{A}, z \in \mathcal{A} \in X $ abgeschlossen mit $z \in X \setminus A$
Dann gilt:
\begin{equation*}
    \begin{split}
        \exists V \subseteq \mathcal{U}(z) \text{ offen } &: V \cap A = \emptyset \wedge\\
        \forall \epsilon > 0 \exists f \in \mathcal{A} &: \forall x \in X : 0 \leq f(x) \leq 1\\
        \forall x \in V &: f(x) \leq \epsilon,\\
        \forall x \in A &: f(x) \geq 1-\epsilon
    \end{split}
\end{equation*}
Weiter zum Beweis des Lemmas:
\begin{proof}{Lemma 9.0.7}
    
    \begin{enumerate}
        \item[(1.)] $\exists h \in \mathcal{A}, \delta \in [0,1]$ $V \in \mathcal{U}(z)$  offen:
        \begin{equation*}
            \begin{split}
                \forall x \in X &: 0 \leq h(x) \leq 1\\
                \forall x \in V &: h(x) \leq \delta/3\\
                \forall x \in A &: h(x) \geq \delta\\
                h(z) &= 0
            \end{split}
        \end{equation*}
        Für den rest des schrittes verweise ich auf die Vorlesung vom 2025.11.03

        \item[(2.)] Sei $k \in \bbN : k-1 \leq \frac{1}{\delta} \leq k$\\
    Für $n \in \mathbb{N}$ sei $p_n:=[1-k^n]^{k^n} \Rightarrow p \in \mathcal{A}. p_n(z) = 1. \forall x \in X:0\leq p_n(x)\leq 1$
    \begin{itemize}
        \item[$x \in A$:]
        \begin{equation*}
            \begin{split}
                h(x)^n & < (\frac{\delta}{3})^n \Rightarrow -h(x)^n \geq -(\frac{\delta}{3})^n > -1\\
                \overset{\text{Bernulli}}{\Rightarrow} (1-h(x)^n)^{k^n} 
                &\geq 1-k^n(\frac{\delta}{3})^n \\
                &\geq 1-k^n(\frac{1}{3k})^n = 1-(\frac{1}{3})^n\\
                k-1 &\leq \frac{1}{\delta}, \delta k \leq 1+\delta \leq z\\
                \Rightarrow \lim_{n\to\infty} p_n(x) &= 1 \quad \text{ für alle x in V gleichmäßig}
            \end{split}
        \end{equation*}
        \item[$x \in V$:]
        \begin{equation*}
            \begin{split}
                k h(x) &\leq k \delta > 1\\
                \overset{\text{Bernulli}}{\Rightarrow} (1-k^n h(x)^n) &\leq (1-h(x)^n)^{k^n} \\
                \Rightarrow 0 \leq p_n(x) &= (1-h(x)^n)^{k^n} (1+k^n h(x)^n)\\
                &\leq (1-h(x)^n)^{k^n}(1+k^n h(x)^n)\\
                \leq (1-h(x)^n)^{k^n}(1+k^n h(x)^n) &= \frac{1}{k^n h(x)^n} 
                \underbrace{1-h(x)^2n^{k^n}}_{\leq 1}\\
                \leq \frac{1}{k^nh(x)} \leq (\frac{1}{k \delta})^n\\
                \Rightarrow \lim_{n\to\infty} p_n(x) &= 0 \quad \text{ für alle x in A gleichmäßig}
            \end{split}
        \end{equation*}
        Also erhalten wir: für n hinreichend gorss gilt: $p_n(x) \leq \epsilon$ für alle $x \in A$ 
        und $p_n(x) \geq 1-\epsilon$ für alle $x \in V$. \\
        Daraus folt für $f := 1-p_n \in \mathcal{A}$ gilt:
        \begin{equation*}
            \begin{split}
                \forall x \in X : 0 \leq f(x) \leq 1\\
                \forall x \in V : f(x) \leq \epsilon\\
                \forall x \in A : f(x) \geq 1-\epsilon
            \end{split}
        \end{equation*}
    \end{itemize}
    \end{enumerate}
\end{proof}

\begin{corollary}{Um das Lemma anzuwenden}
Sei $\mathcal{A} \subseteq C(X,\mathbb{R})$ Punktetrennende Unteralgebra mit $1 \in \mathcal{A}$, 
$A,B \subseteq X$ Abgeschlossen und disjunkt $A \cap B = \emptyset$. Dann gilt:
$$
\eps > 0 \Rightarrow \exists f \in \mathcal{A} :
\forall x \in X : 0 \leq f(x) \leq 1 \wedge \forall x \in B:
f(x) \geq 1-\epsilon \wedge \forall x \in A : f(x) \leq 1 -\epsilon 
$$
\end{corollary}

\begin{proof}
    $\mathcal{A}= \emptyset$.Dann ist das ganze trivial.
    Sei also $A \neq \emptyset$. Für $a \in A$ Wähle $V_a \in \mathcal{a}$ offen.\\
    Und $V_a \cap B = \emptyset$

    \nt{
        $x,y\in X,x\neq y$ Wähle $f \in \mathcal{A}:f(x)\neq f(y)$
        Dann:
        $$
        f^{-1}((f(x)-\delta,f(x)+\delta)), f^{-1}((f(y)-\delta,f(y)+\delta))
        $$
        Dann wähle $\delta := |\frac{f(x)-f(y)}{3}|$.\\
        Dann sind die beiden Mengen disjunkt. \\
        (Beweis $\langle X,d\rangle$ erfüllt (T2))
    }
    $A \subseteq \bigcup_{a \in A}V_a\Rightarrow$ Wähöe $a_1,...,a_n \in A:
    A \subseteq \bigcup_{i=1}^{n}V_{a_i}$
\end{proof}

\thm{}
{
    Sei $X$ Kompakt, $C(CX,\mathbb{R})$ mit Unteralgebra, nirgends verschwindend und

\begin{proof}{Satz; 10.2.1}\\
    Sei $B=\{f\in \mathcal{A}\mid f(x)\in \mathbb{R}\}\subseteq C(X,\mathbb{R})$
    Unteralgebra von $\mathcal{A}$.\\

    Es gelte: 
  $$
    \begin{cases}
        \forall f \in \mathcal{A}:Re f = \frac{1}{2}(f+\overline{f})\\
        \forall f \in \mathcal{B}: Im f = \frac{1}{2i}(f-\overline{f})
    \end{cases}
 $$
    Sei $x \in X$ Wähle $f \in \mathcal{A}: f(x) \neq f(y) \Rightarrow$ entweder 
    $$
    \begin{cases}
    Re f(x) \neq Re f(y)\\
    \text{oder } Im f(x) \neq Im f(y)
    \end{cases}$$
     $\Rightarrow B$ ist punktetrennend.\\
    Aus dem Korollar folgt nun $\overline{B}^{\infty} = C(X,\mathbb{R}) \Rightarrow
    \overline{\mathcal{A}}^{\infty} = C(X,\mathbb{R})$
    Sei $f \in C(X,\mathbb{C})$ $Re f \in \overline{B}^{\infty} \subseteq \overline{\mathcal{A}}^{\infty} \Rightarrow
    f = Re f + i Im f \in \overline{\mathcal{A}}^{\infty}$
\end{proof}
